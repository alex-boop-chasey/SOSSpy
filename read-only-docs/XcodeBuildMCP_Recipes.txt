12. SIMULATOR RESET (use immediately when state is corrupted)
# Erase all simulator data — fixes stuck notifications, battery %, location, widgets, routing
mcp xcodebuildmcp.erase_sims udid=<simUDID>
# Then reboot fresh
mcp xcodebuildmcp.boot_sim udid=<simUDID>
mcp xcodebuildmcp.open_sim udid=<simUDID>
